To see the effect of controllers on ROCK4 using Pleiades Model

using OrdinaryDiffEq, ODE, ODEInterfaceDiffEq, LSODA, Sundials, DiffEqDevTools

f = (du,u,p,t) -> begin
  @inbounds begin
  x = view(u,1:7)   # x
  y = view(u,8:14)  # y
  v = view(u,15:21) # x′
  w = view(u,22:28) # y′
  du[1:7] .= v
  du[8:14].= w
  for i in 14:28
    du[i] = zero(u[1])
  end
  for i=1:7,j=1:7
    if i != j
      r = ((x[i]-x[j])^2 + (y[i] - y[j])^2)^(3/2)
      du[14+i] += j*(x[j] - x[i])/r
      du[21+i] += j*(y[j] - y[i])/r
    end
  end
  end
end

prob = ODEProblem(f,[3.0,3.0,-1.0,-3.0,2.0,-2.0,2.0,3.0,-3.0,2.0,0,0,-4.0,4.0,0,0,0,0,0,1.75,-1.5,0,0,0,-1.25,1,0,0],(0.0,3.0))

abstols = 1.0 ./ 10.0 .^ (6:9)
reltols = 1.0 ./ 10.0 .^ (3:6);

using Plots; gr()


sol = solve(prob,Vern8(),abstol=1/10^12,reltol=1/10^10,maxiters=1000000)
test_sol = TestSolution(sol);
plot(sol)

PI with .9 is default. Authors have suggested Predictive control with gamma 0.8

Names  = ["ROCK4 PI .9" "ROCK4 PI .8" "ROCK4 Pred .9" "ROCK4 Pred .8" "Tsit5"]
setups = [
          Dict(:alg=>ROCK4(controller=:PI),:gamma=>9//10),
          Dict(:alg=>ROCK4(controller=:PI),:gamma=>8//10),
          Dict(:alg=>ROCK4(controller=:Predictive),:gamma=>9//10),
          Dict(:alg=>ROCK4(controller=:Predictive),:gamma=>8//10),
          Dict(:alg=>Tsit5())
]

5-element Array{Dict{Symbol,V} where V,1}:
 Dict{Symbol,Any}(:gamma=>9//10,:alg=>ROCK4(0, 152, :PI))        
 Dict{Symbol,Any}(:gamma=>4//5,:alg=>ROCK4(0, 152, :PI))

wp = WorkPrecisionSet(prob,abstols,reltols,setups;names=Names,appxsol=test_sol,save_everystep=false,numruns=10,maxiters=10000)
plot(wp,dpi=200,linewidth=1.0,legend=:bottomleft,legendfontsize=4)

wp = WorkPrecisionSet(prob,abstols,reltols,setups;names=Names,appxsol=test_sol,numruns=10,maxiters=10000,error_estimate=:L2,dense_errors=true)
plot(wp,dpi=200,linewidth=1.0,legend=:bottomleft,legendfontsize=4)
