To see the effect of controllers on ROCK4 using Lotka Volterra Model

using OrdinaryDiffEq, ParameterizedFunctions, DiffEqDevTools

f = @ode_def LotkaVolterra begin
  dx = a*x - b*x*y
  dy = -c*y + d*x*y
end a b c d

p = [1.5,1.0,3.0,1.0]
prob = ODEProblem(f,[1.0;1.0],(0.0,10.0),p)

abstols = 1.0 ./ 10.0 .^ (6:13)
reltols = 1.0 ./ 10.0 .^ (3:10);
sol = solve(prob,Vern7(),abstol=1/10^14,reltol=1/10^14)
test_sol = TestSolution(sol)
using Plots; gr()
plot(sol)

PI with .9 is default. Authors have suggested Predictive control with gamma 0.8

Names  = ["ROCK4 PI .9" "ROCK4 PI .8" "ROCK4 Pred .9" "ROCK4 Pred .8" "Tsit5"]
setups = [
          Dict(:alg=>ROCK4(controller=:PI),:gamma=>9//10),
          Dict(:alg=>ROCK4(controller=:PI),:gamma=>8//10),
          Dict(:alg=>ROCK4(controller=:Predictive),:gamma=>9//10),
          Dict(:alg=>ROCK4(controller=:Predictive),:gamma=>8//10),
          Dict(:alg=>Tsit5())
]

5-element Array{Dict{Symbol,V} where V,1}:
 Dict{Symbol,Any}(:gamma=>9//10,:alg=>ROCK4(0, 152, :PI))        
 Dict{Symbol,Any}(:gamma=>4//5,:alg=>ROCK4(0, 152, :PI))         
 Dict{Symbol,Any}(:gamma=>9//10,:alg=>ROCK4(0, 152, :Predictive))
 Dict{Symbol,Any}(:gamma=>4//5,:alg=>ROCK4(0, 152, :Predictive)) 
 Dict(:alg=>Tsit5())

wp = WorkPrecisionSet(prob,abstols,reltols,setups;names=Names,appxsol=test_sol,save_everystep=false,numruns=10,maxiters=Int(5e4))
plot(wp,dpi=200,linewidth=1.0,legend=:bottomleft,legendfontsize=4)

wp = WorkPrecisionSet(prob,abstols,reltols,setups;names=Names,appxsol=test_sol,numruns=10,maxiters=Int(5e4),error_estimate=:L2,dense_errors=true)
plot(wp,dpi=200,linewidth=1.0,legend=:bottomleft,legendfontsize=4)

All controller setups are performing equally.
