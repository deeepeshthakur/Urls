To see the effect of controllers on ROCK2 using FitzhughNagumo Model

using OrdinaryDiffEq, ParameterizedFunctions, DiffEqDevTools, Plots

f = @ode_def FitzhughNagumo begin
  dv = v - v^3/3 -w + l
  dw = τinv*(v +  a - b*w)
end a b τinv l

p = [0.7,0.8,1/12.5,0.5]
prob = ODEProblem(f,[1.0;1.0],(0.0,10.0),p)

abstols = 1.0 ./ 10.0 .^ (6:13)
reltols = 1.0 ./ 10.0 .^ (3:10);

sol = solve(prob,Vern7(),abstol=1/10^14,reltol=1/10^14)
test_sol = TestSolution(sol)
gr()
plot(sol)

Different controllers with different gamma values. PI with .9 is default. Authors have suggested Predictive control with gamma 0.8

Names  = ["ROCK2 PI .9" "ROCK2 PI .8" "ROCK2 Pred .9" "ROCK2 Pred .8" "Tsit5"]
setups = [
          Dict(:alg=>ROCK2(controller=:PI),:gamma=>9//10),
          Dict(:alg=>ROCK2(controller=:PI),:gamma=>8//10),
          Dict(:alg=>ROCK2(controller=:Predictive),:gamma=>9//10),
          Dict(:alg=>ROCK2(controller=:Predictive),:gamma=>8//10),
          Dict(:alg=>Tsit5())
]

5-element Array{Dict{Symbol,V} where V,1}:
 Dict{Symbol,Any}(:gamma=>9//10,:alg=>ROCK2(0, 200, :PI))        
 Dict{Symbol,Any}(:gamma=>4//5,:alg=>ROCK2(0, 200, :PI))         
 Dict{Symbol,Any}(:gamma=>9//10,:alg=>ROCK2(0, 200, :Predictive))
 Dict{Symbol,Any}(:gamma=>4//5,:alg=>ROCK2(0, 200, :Predictive))

wp = WorkPrecisionSet(prob,abstols,reltols,setups;names=Names,appxsol=test_sol,save_everystep=false,numruns=10,maxiters=Int(5e5))
plot(wp,dpi=200,linewidth=1.0,legend=:bottomleft,legendfontsize=4)
