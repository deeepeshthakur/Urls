Comparing controllers for ROCK2 and ROCK4

using OrdinaryDiffEq, ParameterizedFunctions, Plots, DiffEqDevTools
gr() #gr(fmt=:png)

f = @ode_def Hires begin
  dy1 = -1.71*y1 + 0.43*y2 + 8.32*y3 + 0.0007
  dy2 = 1.71*y1 - 8.75*y2
  dy3 = -10.03*y3 + 0.43*y4 + 0.035*y5
  dy4 = 8.32*y2 + 1.71*y3 - 1.12*y4
  dy5 = -1.745*y5 + 0.43*y6 + 0.43*y7
  dy6 = -280.0*y6*y8 + 0.69*y4 + 1.71*y5 -
           0.43*y6 + 0.69*y7
  dy7 = 280.0*y6*y8 - 1.81*y7
  dy8 = -280.0*y6*y8 + 1.81*y7
end

u0 = zeros(8)
u0[1] = 1
u0[8] = 0.0057
prob = ODEProblem(f,u0,(0.0,321.8122))

sol = solve(prob,Rodas5(),abstol=1/10^14,reltol=1/10^14)
test_sol = TestSolution(sol)
abstols = 1.0 ./ 10.0 .^ (4:11)
reltols = 1.0 ./ 10.0 .^ (1:8);

plot(sol)

High Tolerances

This is the speed when you just want the answer.

PI and predictive controllers are compared. Standard is not considered. Also 2 Gamma values for Predictive control are used because the suthors suggested Predictive cotrol with gamma 0.8, while default gamma is 0.9

abstols = 1.0 ./ 10.0 .^ (5:8)
reltols = 1.0 ./ 10.0 .^ (1:4);
Names = ["Rosen23" "ROCK2 PI" "ROCK2 Pred" "ROCK4 PI" "ROCK4 Pred 0.9" "ROCK4 Pred 0.8"]
setups = [
          Dict(:alg=>Rosenbrock23()),
          Dict(:alg=>ROCK2(controller=:PI)),
          Dict(:alg=>ROCK2(controller=:Predictive)),
          Dict(:alg=>ROCK4(controller=:PI)),
          Dict(:alg=>ROCK4(controller=:Predictive)),
          Dict(:alg=>ROCK4(controller=:Predictive),:gamma=>8//10)
          ]
wp = WorkPrecisionSet(prob,abstols,reltols,setups;names=Names,
                      save_everystep=false,appxsol=test_sol,maxiters=Int(5e5),numruns=20)
plot(wp,linewidth=1.0,dpi=200,legend=:bottomright,legendfontsize=4)

wp = WorkPrecisionSet(prob,abstols,reltols,setups;names=Names,dense = false,verbose=false,
                      appxsol=test_sol,maxiters=Int(5e5),error_estimate=:l2)
plot(wp,linewidth=1.0,dpi=200,legend=:bottomright,legendfontsize=4)

wp = WorkPrecisionSet(prob,abstols,reltols,setups;names=Names,
                      appxsol=test_sol,maxiters=Int(5e5),error_estimate=:L2)
plot(wp,linewidth=1.0,dpi=200,legend=:bottomright,legendfontsize=4)
