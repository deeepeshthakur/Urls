using StochasticDiffEq, DiffEqDevTools, Test, LinearAlgebra, Random, Plots
Random.seed!(100)

const σ_const = 0.87
const μ = 1.01

u0 = rand(2)
A = ones(2)
B = σ_const*ones(2)

function f_(du,u,p,t)
  du .= (t^2)
end

function f_analytic(u0,p,t,W)
  u0 + ((t^3)/3)*A + B*sum(W)
end

function σ(du,u,p,t)
  du[1,1] = σ_const  #*u[1]
  du[1,2] = σ_const  #*u[1]
  du[2,1] = σ_const  #*u[2]
  du[2,2] = σ_const  #*u[2]
end

ff = SDEFunction(f_,σ,analytic=f_analytic)

prob = SDEProblem(ff,σ,u0,(0.0,1.0),noise_rate_prototype=rand(2,2))
dts = (1/2) .^ (10:-1:3)
