Temp test to check the performance after introducing step_size limitations to prevent solver from getting unstable 

using OrdinaryDiffEq, ParameterizedFunctions, Plots, DiffEqDevTools
gr() #gr(fmt=:png)

f = @ode_def Hires begin
  dy1 = -1.71*y1 + 0.43*y2 + 8.32*y3 + 0.0007
  dy2 = 1.71*y1 - 8.75*y2
  dy3 = -10.03*y3 + 0.43*y4 + 0.035*y5
  dy4 = 8.32*y2 + 1.71*y3 - 1.12*y4
  dy5 = -1.745*y5 + 0.43*y6 + 0.43*y7
  dy6 = -280.0*y6*y8 + 0.69*y4 + 1.71*y5 -
           0.43*y6 + 0.69*y7
  dy7 = 280.0*y6*y8 - 1.81*y7
  dy8 = -280.0*y6*y8 + 1.81*y7
end

u0 = zeros(8)
u0[1] = 1
u0[8] = 0.0057
prob = ODEProblem(f,u0,(0.0,321.8122))

sol = solve(prob,Rodas5(),abstol=1/10^14,reltol=1/10^14)
test_sol = TestSolution(sol)

abstols = 1.0 ./ 10.0 .^ (5:8)
reltols = 1.0 ./ 10.0 .^ (1:4);
Names  = ["Rosen23" "Rodas3" "ROCK2" "ROCK4"]
setups = [Dict(:alg=>Rosenbrock23()),
          Dict(:alg=>Rodas3()),
          Dict(:alg=>ROCK2()),
          Dict(:alg=>ROCK4())
          ];

Only a few solvers are compared like Rodas3 Rosenbrock23 and Rodas5

High tol test just to see the speeds 

wp = WorkPrecisionSet(prob,abstols,reltols,setups;names=Names,
                      save_everystep=false,appxsol=test_sol,maxiters=Int(4e5),numruns=10)
plot(wp,dpi=200,linewidth=1.0,legend=:bottomleft,legendfontsize=4)

There is a limiting of stepsize after a while the ROCK solvers sort of become constant step solvers after they reach maximum step_size
