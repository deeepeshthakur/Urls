using OrdinaryDiffEq, DiffEqDevTools, Test, LinearAlgebra, Random
using DiffEqProblemLibrary.ODEProblemLibrary: importodeproblems; importodeproblems()
import DiffEqProblemLibrary.ODEProblemLibrary: prob_ode_linear, prob_ode_2Dlinear
probArr = Vector{ODEProblem}(undef, 2)
probArr[1] = prob_ode_linear
probArr[2] = prob_ode_2Dlinear

using Plots
dts = 1 .//2 .^(8:-1:4)
testTol = 0.1
prob = probArr[1]

ODEProblem with uType Float64 and tType Float64. In-place: false
timespan: (0.0, 1.0)

sim = test_convergence(dts,prob,ROCK2())
@test sim.𝒪est[:l∞] ≈ 2 atol=testTol
plot(sim,dpi=200,size=(600,400),linewidth=1.0,minorgrid=true,gridalpha=0.5,minorgridalpha=0.5)
